# pages/locators.py
import os

class AutomationPage:
    def __init__(self, page):
        self.page = page
        self.url = "https://www.automationanywhere.com/products/enterprise/community-edition"

    def login(self, user, pw):
        self.page.goto(self.url)
        # Selectors based on general Automation Anywhere patterns [cite: 6]
        self.page.fill('input[name="user"]', user)
        self.page.fill('input[name="password"]', pw)
        self.page.click('button[type="submit"]')

    def create_message_box_task(self, bot_name):
        # Use Case 1: Message Box Task [cite: 7, 9]
        self.page.click('text="Automation"')
        self.page.click('button:has-text("Create")')
        self.page.click('text="Task Bot"')
        self.page.fill('input[label="Name*"]', bot_name)
        self.page.click('button:has-text("Create")')
        
        # Add Message Box action [cite: 11]
        self.page.fill('input[placeholder="Search actions"]', "Message Box")
        self.page.dblclick('text="Message Box"')
        
        # Verify and Save [cite: 12]
        self.page.wait_for_selector('.right-panel')
        self.page.click('button:has-text("Save")')

    def create_form_with_upload(self, form_name, file_name):
        # Use Case 2: Form with Upload [cite: 15, 17, 18]
        self.page.click('text="Automation"')
        self.page.click('button:has-text("Create")')
        self.page.click('text="Form"')
        self.page.fill('input[label="Name*"]', form_name)
        self.page.click('button:has-text("Create")')

        # Drag and Drop (Simulated via clicking/adding) [cite: 19]
        self.page.click('text="Textbox"')
        self.page.click('text="Select File"')

        # Upload Flow [cite: 21, 22]
        self.page.fill('input[type="text"]', "Automated Input")
        file_path = os.path.abspath(file_name)
        self.page.set_input_files('input[type="file"]', file_path)
        
        self.page.click('button:has-text("Save")')
